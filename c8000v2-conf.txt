! C8000V-2 (AZ2) - Gi1 10.1.1.90/27 (RouterSubnet2), Gi2 10.1.1.110/27 (ReceiverSubnet2)
! BGP to VPC Route Server Endpoint-2 (10.1.1.71, ASN 65100) with BFD health check;
! standby multicast forwarder (see the ip igmp static-group note on Gi2).
!
hostname c8000v-2
!
ip multicast-routing distributed
!
interface GigabitEthernet1
 description RouterSubnet2-underlay-RS-peering
 ip address 10.1.1.90 255.255.255.224
 bfd interval 300 min_rx 300 multiplier 3
 no shutdown
!
interface GigabitEthernet2
 description ReceiverSubnet2-multicast
 ip address 10.1.1.110 255.255.255.224
 ip pim sparse-mode
 ! STANDBY: leave the static-group disabled. If both routers pull the same
 ! group, every receiver gets duplicate packets (TGW does not dedupe and the
 ! two Gi2 interfaces share no segment for a PIM assert). Enable this only
 ! when C8000V-1 is down, or automate the switchover with an EEM applet.
 ! ip igmp static-group 239.1.1.1
 no shutdown
!
interface Tunnel101
 description GRE-to-OnPrem-Lo0
 ip address 172.16.101.2 255.255.255.252
 ip pim sparse-mode
 tunnel source GigabitEthernet1
 tunnel destination 192.168.255.1
 ip mtu 1476
 ip tcp adjust-mss 1436
!
! GRE destination (on-prem Lo0) via Gi1 gateway -> RouterRouteTable -> TGW -> DXGW
ip route 192.168.255.1 255.255.255.255 10.1.1.65
! Rest of the VPC (other-AZ subnets, EICE) via Gi1 gateway; connected /27s stay preferred
ip route 10.1.1.0 255.255.255.0 10.1.1.65
!
! RP is the on-prem router Lo0
ip pim rp-address 192.168.255.1
! Force PIM RPF for the RP and the source range over the tunnel
! (the unicast route to Lo0 points at the underlay, where PIM does not run)
ip mroute 192.168.255.1 255.255.255.255 Tunnel101
ip mroute 172.20.51.0 255.255.255.0 Tunnel101
!
ip prefix-list ONPREM-LAN seq 5 permit 172.20.51.0/24
ip prefix-list DENY-ALL seq 5 deny 0.0.0.0/0 le 32
!
router bgp 65200
 bgp router-id 10.1.1.90
 bgp log-neighbor-changes

 ! VPC Route Server Endpoint-2 - BFD-based failover
 neighbor 10.1.1.71 remote-as 65100
 neighbor 10.1.1.71 fall-over bfd

 ! On-prem router over the GRE overlay
 neighbor 172.16.101.1 remote-as 65000

 address-family ipv4
  ! Re-advertise the on-prem LAN (learned over the tunnel) to the Route Server
  neighbor 10.1.1.71 activate
  neighbor 10.1.1.71 prefix-list ONPREM-LAN out

  ! Advertise nothing to on-prem (it reaches the VPC via the DX underlay)
  neighbor 172.16.101.1 activate
  neighbor 172.16.101.1 prefix-list DENY-ALL out
 exit-address-family
